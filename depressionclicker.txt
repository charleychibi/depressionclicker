Layout
	use default
		
Buttons
	*spoonButton
		name:Generate Spoons
		desc:Click the button to generate spoons
		on click:anim icon wobble
		on click:yield 1 spoon
		icon:spoonbutton.png
		no text
		class:bigButton hasFlares
		icon class:shadowed
		tooltip origin:bottom
		tooltip class:red
		
Resources
	*spoon|spoons
		name:Spoon|Spoons
		desc:Spoons are symbolic of how much energy you have to do things. Generate spoons and aim for mental wellbeing.
		icon:icons[0,0]
		class:noBackground
		show earned
		
Shinies
	*luckyMeds
		on click:log Woop
		movement:onRight moveLeft fade bounce:0.05
		frequency:30
		frequency variation:20
		icon:stuff/luckyBunny.png
		class:bigButton
		on click:
			$boost=1
			if (have clover) $boost=4
			if (chance(25%))
				//get at least 7, or between 1 and 3 minutes of our spoon production
				$amount=max(7,random(spoons:ps*60*1,spoons:ps*60*3))*$boost
				toast The meds grant you<//><b>[$amount] spoons</b>.
				yield $amount spoons
			else
				$amount=1*$boost
				toast The meds grant you<//><b>[$amount] golden spoons[s?$amount]</b>!
				yield $amount goldenSpoon
			end
		end
